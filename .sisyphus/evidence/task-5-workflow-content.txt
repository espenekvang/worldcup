# GitHub Actions CI/CD Pipeline — Build and Deploy to Azure Container Apps
#
# Required GitHub repository secrets:
#   AZURE_CREDENTIALS      — Azure service principal JSON (from `az ad sp create-for-rbac --sdk-auth`)
#   ACR_NAME               — ACR resource name, e.g. acrworldcup (NOT the full login server URL)
#   RESOURCE_GROUP         — Azure resource group name, e.g. rg-worldcup
#   CONTAINER_APP_NAME     — Container App name, e.g. ca-worldcup
#   VITE_GOOGLE_CLIENT_ID  — Google OAuth client ID (baked into frontend JS bundle at build time)

name: Build and Deploy

on:
  push:
    branches: [main]
  workflow_dispatch:

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Login to Azure
        uses: azure/login@v2
        with:
          creds: ${{ secrets.AZURE_CREDENTIALS }}

      - name: Build and push image to ACR
        run: |
          az acr build \
            --registry ${{ secrets.ACR_NAME }} \
            --image worldcup:${{ github.sha }} \
            --build-arg VITE_GOOGLE_CLIENT_ID=${{ secrets.VITE_GOOGLE_CLIENT_ID }} \
            .

      - name: Deploy to Container App
        run: |
          az containerapp update \
            --name ${{ secrets.CONTAINER_APP_NAME }} \
            --resource-group ${{ secrets.RESOURCE_GROUP }} \
            --image ${{ secrets.ACR_NAME }}.azurecr.io/worldcup:${{ github.sha }}

      - name: Update ingress target port to app port
        run: |
          az containerapp ingress update \
            --name ${{ secrets.CONTAINER_APP_NAME }} \
            --resource-group ${{ secrets.RESOURCE_GROUP }} \
            --target-port 8080
